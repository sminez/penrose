macro_rules! stack {
    ([$($up:expr),*], $focus:expr, [$($down:expr),*]) => { ... };
    ([$($up:expr),*], $focus:expr) => { ... };
    ($focus:expr, [$($down:expr),*]) => { ... };
    ($focus:expr, $($down:expr),+) => { ... };
    ($focus:expr) => { ... };
}
Expand description

Create a Stack containing the arguments. The only required element is the focus, it is not possible to create an empty Stack;

let s = stack!([1, 2], 3, [4, 5]);
let s = stack!([1, 2], 3);
let s = stack!(1, [2, 3]);
let s = stack!(1, 2, 3);
let s = stack!(1);
